Bootstrap Modal

удаляют прокрутку у body чтобы крутился контент в попапе

1 попап в 1 момент времени

Попап

<div id='modal' class="modal" tabindex="-1">

  <div class="modal-dialog modal-dialog-centered modal-dialog-scrollable">
  
    <div class="modal-content">
      
	  <div class="modal-header">
        <h5 class="modal-title">header</h5>
        <button type="button" class="btn-close" data-bs-dismiss="modal" aria-label="Close"></button>
      </div>
	  
      <div class="modal-body">
        <p>body</p>
      </div>
	  
      <div class="modal-footer">
        <button type="button" class="btn btn-secondary" data-bs-dismiss="modal">Close</button>
        <button type="button" class="btn btn-primary">button</button>
      </div>
	  
    </div>
	
  </div>
  
</div>

Кнопка

<button type="button" class="btn btn-primary" data-bs-toggle="modal" data-bs-target="#modal">
  button
</button>

Запуск через яваскрипт

var modal = new bootstrap.Modal(document.getElementById('modal'), options)

с опциями
var modal = new bootstrap.Modal(document.getElementById('modal'), {keyboard: false})

new bootstrap.Modal(document.getElementById('modal')).show();

Все методы асинхронны
myModal.toggle()
myModal.show()
myModal.hide()
myModal.dispose() удалить из DOM

Инстанс попапа
var modalDiv = document.getElementById('modal')
var modal = bootstrap.Modal.getInstance(modalDiv)
